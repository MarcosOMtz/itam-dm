  for(var in vars.categoricas){
    tab <- prop.table(table(ds[,var]))
    noms <- names(tab)
    props <- as.numeric(tab)
    cambia <- function(x){
      ind = which(noms == x)
      return(props[ind])
    }
    ds[,paste(var,'.num',sep='')] <- apply(as.data.frame(ds[,var]),1,cambia)
  } 
  return(ds)
}
